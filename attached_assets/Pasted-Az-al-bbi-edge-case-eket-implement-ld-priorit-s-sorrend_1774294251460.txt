PRIORITÁS 1 – ÉLETVESZÉLYES EDGE CASE-EK
A) Telefon leesés vs valódi baleset megkülönböztetése
javascript// A jelenlegi detekció csak G-erőt néz.
// Add hozzá ezt a logikát MINDEN baleset triggerhez:

function isPhoneDrop(sensorData) {
  const { gForce, speedKmh, speedChangeMs } = sensorData;
  
  // Ha a GPS szerint a motoros még mozog hasonló sebességen
  // de a telefon G-ereje magas → telefon esett le, nem baleset
  if (gForce > 3.0 && speedChangeMs < 5 && speedKmh > 10) {
    return true; // false positive – telefon leesett
  }
  return false;
}

// Módosítsd az accident detection logikát:
if (isPhoneDrop(sensorData)) {
  console.log('Phone drop detected – ignoring');
  return; // ne triggerelj semmit
}

function buildLowBatteryAlert(name, location, percent) {
  const mapsLink = `https://maps.google.com/?q=${location.lat},${location.lon}`;
  return `⚡ MOTOGUARD LOW BATTERY ALERT
Rider: ${name}
Battery: ${percent}% – phone may go offline soon
Last known location: ${mapsLink}
Time: ${new Date().toLocaleString()}
This is an automated safety alert from MotoGuard.`;
}

H) Idle detection – tankolás, pihenő automatikus felismerése
javascript// Ha a motoros 3+ percig áll menet közben:
const IDLE_THRESHOLD_MS = 3 * 60 * 1000; // 3 perc

useEffect(() => {
  let idleTimer;
  
  if (isRiding && currentSpeed < 2) {
    idleTimer = setTimeout(() => {
      setRideStatus('paused');
      speak("Ride paused. Tap to resume when ready.");
      
      // Ride timer megáll, track rögzítés szünetel
      pauseRideTracking();
    }, IDLE_THRESHOLD_MS);
  } else if (currentSpeed >= 2) {
    clearTimeout(idleTimer);
    
    if (rideStatus === 'paused') {
      setRideStatus('active');
      resumeRideTracking();
      speak("Ride resumed");
    }
  }
  
  return () => clearTimeout(idleTimer);
}, [currentSpeed, isRiding]);

PRIORITÁS 3 – KOMFORT FEJLESZTÉSEK
I) GPS elvesztés kezelése alagútban
javascript// Ha GPS signal elvész (pontosság > 50m vagy nincs fix):
function handleGPSLoss(lastKnownLocation, lastSpeed, lastHeading) {
  setGpsStatus('lost');
  
  // Dead reckoning: becsüljük a pozíciót sebesség + irány alapján
  const estimatedPosition = estimatePositionByDeadReckoning(
    lastKnownLocation,
    lastSpeed,
    lastHeading,
    timeSinceLastFix
  );
  
  // Térképen jelezd hogy becslés:
  setPositionAccuracy('estimated');
  updateMapPosition(estimatedPosition);
  
  // 30 másodperc után figyelmeztetés:
  if (timeSinceLastFix > 30000) {
    speak("GPS signal lost. Navigation may be inaccurate.");
  }
}

function estimatePositionByDeadReckoning(pos, speedMs, headingDeg, elapsedMs) {
  const distanceM = speedMs * (elapsedMs / 1000);
  const headingRad = headingDeg * Math.PI / 180;
  
  const deltaLat = (distanceM * Math.cos(headingRad)) / 111320;
  const deltaLon = (distanceM * Math.sin(headingRad)) / 
    (111320 * Math.cos(pos.latitude * Math.PI / 180));
  
  return {
    latitude: pos.latitude + deltaLat,
    longitude: pos.longitude + deltaLon,
    isEstimated: true
  };
}
J) Fekvőrendőr / útgödör false positive szűrése
javascript// Fekvőrendőr jellemzői:
// - Hirtelen G-erő spike (< 0.3 másodperc)
// - Utána azonnal visszaáll normálra
// - Sebesség nem csökkent jelentősen
// - Ismétlődik (ha az úton több van)

function isSpeedBump(accelerationHistory) {
  const spike = accelerationHistory.slice(-5); // utolsó 500ms
  const duration = spike.filter(a => a > 2.0).length;
  
  // Ha a spike rövid (< 300ms) és a sebesség folytatódott:
  if (duration <= 3 && currentSpeed > lastSpeed * 0.8) {
    return true; // valószínűleg fekvőrendőr
  }
  return false;
}

// Az accident detection-ben:
if (isSpeedBump(recentAcceleration)) {
  console.log('Speed bump detected – ignoring');
  return;
}
