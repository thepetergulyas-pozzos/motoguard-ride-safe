javascriptimport { Accelerometer } from 'expo-sensors';

// 10Hz frissítési ráta (100ms)
Accelerometer.setUpdateInterval(100);

const subscription = Accelerometer.addListener(({ x, y, z }) => {
  const gForce = Math.sqrt(x*x + y*y + z*z);
  
  // MAIDS standard küszöbértékek:
  if (gForce > 3.0) triggerWarning(gForce);
  if (gForce > 4.5) triggerAlert(gForce);
  if (gForce > 6.0) triggerEmergency(gForce);
});

5. SMS KÜLDÉS (valódi)
javascriptimport * as SMS from 'expo-sms';

async function sendEmergencySMS(contacts, message) {
  const isAvailable = await SMS.isAvailableAsync();
  
  if (isAvailable) {
    await SMS.sendSMSAsync(
      contacts.map(c => c.phone),
      message
    );
  } else {
    // Fallback: Linking.openURL('sms:...')
    const phoneNumbers = contacts.map(c => c.phone).join(',');
    await Linking.openURL(`sms:${phoneNumbers}?body=${encodeURIComponent(message)}`);
  }
}

// SMS szöveg formátum:
function buildEmergencySMS(riderName, lat, lon, level) {
  const mapsLink = `https://maps.google.com/?q=${lat},${lon}`;
  return `🚨 MOTOGUARD EMERGENCY ALERT
Rider: ${riderName}
Level: ${level}
Location: ${mapsLink}
Coordinates: ${lat.toFixed(5)}, ${lon.toFixed(5)}
Time: ${new Date().toLocaleString()}
Sent by MotoGuard Safety App`;
}

6. VÉSZJELZÉS PROTOKOLL (3 szint)
Implementáld a teljes accident detection flow-t:
javascript// 1. szint – WARNING (sárga)
async function triggerWarning(gForce, speed) {
  await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Warning);
  showWarningOverlay({
    countdown: 15,
    message: "Are you OK?",
    onConfirm: () => dismissAlert(),
    onTimeout: () => triggerAlert(gForce, speed)
  });
}

// 2. szint – ALERT (piros)  
async function triggerAlert(gForce, speed) {
  await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Error);
  showAlertOverlay({
    countdown: 30,
    message: "Accident detected! Cancel or help will be called.",
    onConfirm: () => dismissAlert(),
    onTimeout: () => triggerEmergency(gForce, speed)
  });
}

// 3. szint – EMERGENCY (automatikus SMS)
async function triggerEmergency(gForce, speed) {
  const location = await Location.getCurrentPositionAsync({});
  const { latitude, longitude } = location.coords;
  
  // SMS automatikusan (user approval nem kell 3. szintnél)
  const contacts = getEmergencyContacts();
  const sms = buildEmergencySMS(getRiderName(), latitude, longitude, "EMERGENCY");
  await sendEmergencySMS(contacts, sms);
  
  // 112 hívás – Linking (user interaction kell iOS-en)
  showEmergencyCallPrompt({
    script: `This is an automated alert. Rider ${getRiderName()} may have had an accident at ${latitude}, ${longitude}.`,
    onCall: () => Linking.openURL('tel:112')
  });
}

7. ELŐFIZETÉSI RENDSZER (RevenueCat)
javascriptimport Purchases from 'react-native-purchases';

// Inicializálás (App.js-ben):
await Purchases.configure({
  apiKey: Platform.OS === 'ios' 
    ? 'appl_XXXX'   // ← RevenueCat iOS kulcs
    : 'goog_XXXX'   // ← RevenueCat Android kulcs
});

// Csomag definíciók (RevenueCat dashboard-on is beállítandó):
// - com.motoguard.basic.monthly    $2.99
// - com.motoguard.pro.monthly      $5.99
// - com.motoguard.prodrone.monthly $12.99

// Vásárlás:
async function purchasePlan(packageId) {
  try {
    const offerings = await Purchases.getOfferings();
    const pkg = offerings.current.availablePackages
      .find(p => p.product.identifier === packageId);
    
    const { customerInfo } = await Purchases.purchasePackage(pkg);
    return customerInfo.entitlements.active;
  } catch (e) {
    if (!e.userCancelled) console.error(e);
  }
}

// Entitlement check:
async function checkSubscription() {
  const info = await Purchases.getCustomerInfo();
  if (info.entitlements.active['pro_drone']) return 'pro_drone';
  if (info.entitlements.active['pro']) return 'pro';
  if (info.entitlements.active['basic']) return 'basic';
  return 'free';
}

8. HÁTTÉR FUTTATÁS (Keep Awake + Background Task)
javascriptimport { activateKeepAwakeAsync, deactivateKeepAwake } from 'expo-keep-awake';
import * as TaskManager from 'expo-task-manager';
import * as BackgroundFetch from 'expo-background-fetch';

const BACKGROUND_ACCIDENT_TASK = 'background-accident-detection';

// Háttér task regisztrálás:
TaskManager.defineTask(BACKGROUND_ACCIDENT_TASK, async () => {
  // Accelerometer polling háttérben
  return BackgroundFetch.BackgroundFetchResult.NewData;
});

// Menet indításkor:
async function startRide() {
  await activateKeepAwakeAsync();
  await BackgroundFetch.registerTaskAsync(BACKGROUND_ACCIDENT_TASK, {
    minimumInterval: 1, // 1 másodperc
    stopOnTerminate: false,
    startOnBoot: false,
  });
}

// Menet leállításkor:
async function stopRide() {
  deactivateKeepAwake();
  await BackgroundFetch.unregisterTaskAsync(BACKGROUND_ACCIDENT_TASK);
}
