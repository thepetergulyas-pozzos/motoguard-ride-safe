javascriptimport { Audio } from 'expo-av';

// Folyamatos háttér zajszint mérés:
async function monitorAmbientNoise() {
  const recording = new Audio.Recording();
  await recording.prepareToRecordAsync({
    android: { ...Audio.RecordingOptionsPresets.LOW_QUALITY },
    ios: { ...Audio.RecordingOptionsPresets.LOW_QUALITY },
  });
  
  // Decibel szint figyelés:
  recording.setOnRecordingStatusUpdate((status) => {
    const db = status.metering; // negatív szám, pl. -30 dB
    
    // Háttérzaj alap: sisak/motor zaja ~65-75 dB
    // Emberi hang: általában 15-20 dB-vel magasabb mint a háttér
    // Ha a különbség > 15 dB → valaki beszél
    if (db > ambientNoiseLevel + 15) {
      activateVoiceRecognition();
    }
  });
}

// Zajszint kalibrálás induláskor (5 másodperc):
async function calibrateNoise() {
  // 5 másodpercig mér és eltárolja az átlagos háttérzajt
